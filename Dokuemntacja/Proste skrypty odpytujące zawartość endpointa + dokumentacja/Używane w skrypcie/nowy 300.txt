@router.post("/serp/google/organic/live/advanced")
async def get_google_organic_serp_live_advanced(data: SerpOrganicInput):
    """
    Pobiera dane z Live Google Organic SERP Advanced endpoint.
    Zwraca kompletny przeglƒÖd wynik√≥w SERP w czasie rzeczywistym.
    """
    if not DFS_LOGIN or not DFS_PASSWORD:
        logger.error("‚ùå Brak danych logowania do API")
        raise HTTPException(status_code=500, detail="Brak danych logowania do API DataForSEO.")
    
    logger.info(f"‚û°Ô∏è Zapytanie SERP dla s≈Çowa kluczowego: '{data.keyword}'")
    logger.info(f"üìç Lokalizacja: {data.location_code}, Jƒôzyk: {data.language_code}")
    logger.info(f"üñ•Ô∏è UrzƒÖdzenie: {data.device}, OS: {data.os}")
    
    config = dfs_config.Configuration(username=DFS_LOGIN, password=DFS_PASSWORD)
    
    # Przygotowanie requestu
    request_data = [
        SerpGoogleOrganicLiveAdvancedRequestInfo(
            keyword=data.keyword,
            location_code=data.location_code,
            language_code=data.language_code,
            device=data.device,
            os=data.os,
            depth=data.depth,
            calculate_rectangles=data.calculate_rectangles,
            group_organic_results=data.group_organic_results
        )
    ]
    
    try:
        with dfs_api_provider.ApiClient(config) as api_client:
            api_instance = SerpApi(api_client)
            logger.debug("‚û°Ô∏è Wysy≈Çanie ≈ºƒÖdania do DataForSEO SERP API (Google Organic Live Advanced)...")
            
            # Wywo≈Çanie endpointu
            api_response = api_instance.google_organic_live_advanced(request_data)
            
            if not api_response.tasks:
                logger.warning("‚ö†Ô∏è Brak zada≈Ñ w odpowiedzi API.")
                raise HTTPException(status_code=404, detail="Brak danych w odpowiedzi API.")
            
            task = api_response.tasks[0]
            
            if task.status_code != 20000:
                logger.error(f"‚ùå B≈ÇƒÖd zadania: {task.status_code} - {task.status_message}")
                raise HTTPException(
                    status_code=400, 
                    detail=f"B≈ÇƒÖd zadania DataForSEO: {task.status_message}"
                )
            
            if not task.result:
                logger.warning("‚ö†Ô∏è Brak wynik√≥w dla podanego s≈Çowa kluczowego.")
                raise HTTPException(
                    status_code=404, 
                    detail="Brak wynik√≥w SERP dla podanego s≈Çowa kluczowego."
                )
            
            result = task.result[0]
            
            # Logowanie podstawowych informacji o wynikach
            logger.info(f"‚úÖ Pobrano dane SERP dla '{result.keyword}'")
            logger.info(f"üìä Liczba wynik√≥w SERP: {result.se_results_count}")
            logger.info(f"üìã Liczba element√≥w: {result.items_count}")
            logger.info(f"üîó URL sprawdzajƒÖcy: {result.check_url}")
            
            if result.item_types:
                logger.info(f"üìù Typy element√≥w SERP: {', '.join(result.item_types)}")
            
            return {
                "task_info": {
                    "id": task.id,
                    "status_code": task.status_code,
                    "status_message": task.status_message,
                    "cost": task.cost,
                    "execution_time": task.time
                },
                "result": result.to_dict()
            }
    
    except Exception as e:
        logger.exception("‚ùå B≈ÇƒÖd podczas pobierania danych z DataForSEO SERP API")
        raise HTTPException(
            status_code=500, 
            detail=f"B≈ÇƒÖd serwera podczas pobierania danych SERP:\n{str(e)}"
        )

@router.post("/serp/google/organic/live/advanced/filtered")
async def get_filtered_serp_data(data: SerpOrganicInput, element_types: list[str]):
    """
    Pobiera dane SERP i filtruje tylko wybrane typy element√≥w.
    
    Przyk≈Çadowe typy element√≥w:
    - organic: organiczne wyniki wyszukiwania
    - paid: p≈Çatne reklamy
    - featured_snippet: wyr√≥≈ºnione fragmenty
    - people_also_ask: ludzie pytajƒÖ te≈º
    - knowledge_graph: graf wiedzy
    - local_pack: wyniki lokalne
    - images: obrazy
    - shopping: zakupy
    - top_stories: najwa≈ºniejsze wiadomo≈õci
    """
    # Wywo≈Çaj g≈Ç√≥wny endpoint
    full_response = await get_google_organic_serp_live_advanced(data)
    
    # Filtruj elementy wed≈Çug typu
    if 'result' in full_response and 'items' in full_response['result']:
        filtered_items = [
            item for item in full_response['result']['items'] 
            if item.get('type') in element_types
        ]
        
        full_response['result']['items'] = filtered_items
        full_response['result']['items_count'] = len(filtered_items)
        
        logger.info(f"üîç Przefiltrowano do {len(filtered_items)} element√≥w typ√≥w: {', '.join(element_types)}")
    
    return full_response
