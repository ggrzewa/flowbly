# Logger

DFS_LOGIN = os.getenv("DATAFORSEO_LOGIN")
DFS_PASSWORD = os.getenv("DATAFORSEO_PASSWORD")

# Model danych wejściowych dla AI Summary
class SerpAiSummaryInput(BaseModel):
    task_id: str  # ID zadania z poprzedniego zapytania SERP
    prompt: Optional[str] = "Summarize the main findings and key information from these search results"
    support_extra: Optional[bool] = True  # uwzględnianie dodatkowych elementów SERP
    fetch_content: Optional[bool] = False  # pobieranie zawartości stron
    include_links: Optional[bool] = True  # dołączanie linków w podsumowaniu

# Model danych wejściowych dla SERP
class SerpOrganicInput(BaseModel):
    keyword: str
    location_code: Optional[int] = 2616  # Polska
    language_code: Optional[str] = "pl"
    device: Optional[str] = "desktop"  # desktop lub mobile
    os: Optional[str] = "windows"  # windows, macos, android, ios
    depth: Optional[int] = 100  # liczba wyników (max 700)
    calculate_rectangles: Optional[bool] = False  # kalkulacja pozycji pixelowych
    group_organic_results: Optional[bool] = True  # grupowanie powiązanych wyników

@router.post("/serp/google/organic/live/advanced")
async def get_google_organic_serp_live_advanced(data: SerpOrganicInput):
    """
    Pobiera dane z Live Google Organic SERP Advanced endpoint.
    Zwraca kompletny przegląd wyników SERP w czasie rzeczywistym.
    """
    if not DFS_LOGIN or not DFS_PASSWORD:
        logger.error("❌ Brak danych logowania do API")
        raise HTTPException(status_code=500, detail="Brak danych logowania do API DataForSEO.")
    
    logger.info(f"➡️ Zapytanie SERP dla słowa kluczowego: '{data.keyword}'")
    logger.info(f"📍 Lokalizacja: {data.location_code}, Język: {data.language_code}")
    logger.info(f"🖥️ Urządzenie: {data.device}, OS: {data.os}")
    
    config = dfs_config.Configuration(username=DFS_LOGIN, password=DFS_PASSWORD)
    
    # Przygotowanie requestu
    request_data = [
        SerpGoogleOrganicLiveAdvancedRequestInfo(
            keyword=data.keyword,
            location_code=data.location_code,
            language_code=data.language_code,
            device=data.device,
            os=data.os,
            depth=data.depth,
            calculate_rectangles=data.calculate_rectangles,
            group_organic_results=data.group_organic_results
        )
    ]
    
    try:
        with dfs_api_provider.ApiClient(config) as api_client:
            api_instance = SerpApi(api_client)
            logger.debug("➡️ Wysyłanie żądania do DataForSEO SERP API (Google Organic Live Advanced)...")
            
            # Wywołanie endpointu
            api_response = api_instance.google_organic_live_advanced(request_data)
            
            if not api_response.tasks:
                logger.warning("⚠️ Brak zadań w odpowiedzi API.")
                raise HTTPException(status_code=404, detail="Brak danych w odpowiedzi API.")
            
            task = api_response.tasks[0]
            
            if task.status_code != 20000:
                logger.error(f"❌ Błąd zadania: {task.status_code} - {task.status_message}")
                raise HTTPException(
                    status_code=400, 
                    detail=f"Błąd zadania DataForSEO: {task.status_message}"
                )
            
            if not task.result:
                logger.warning("⚠️ Brak wyników dla podanego słowa kluczowego.")
                raise HTTPException(
                    status_code=404, 
                    detail="Brak wyników SERP dla podanego słowa kluczowego."
                )
            
            result = task.result[0]
            
            # Logowanie podstawowych informacji o wynikach
            logger.info(f"✅ Pobrano dane SERP dla '{result.keyword}'")
            logger.info(f"📊 Liczba wyników SERP: {result.se_results_count}")
            logger.info(f"📋 Liczba elementów: {result.items_count}")
            logger.info(f"🔗 URL sprawdzający: {result.check_url}")
            
            if result.item_types:
                logger.info(f"📝 Typy elementów SERP: {', '.join(result.item_types)}")
            
            return {
                "task_info": {
                    "id": task.id,
                    "status_code": task.status_code,
                    "status_message": task.status_message,
                    "cost": task.cost,
                    "execution_time": task.time
                },
                "result": result.to_dict()
            }
    
    except Exception as e:
        logger.exception("❌ Błąd podczas pobierania danych z DataForSEO SERP API")
        raise HTTPException(
            status_code=500, 
            detail=f"Błąd serwera podczas pobierania danych SERP:\n{str(e)}"
        )

@router.post("/serp/ai-summary")
async def get_serp_ai_summary(data: SerpAiSummaryInput):
    """
    Generuje AI podsumowanie wyników SERP na podstawie task_id.
    Wymaga wcześniejszego wykonania zapytania SERP i podania jego task_id.
    """
    if not DFS_LOGIN or not DFS_PASSWORD:
        logger.error("❌ Brak danych logowania do API")
        raise HTTPException(status_code=500, detail="Brak danych logowania do API DataForSEO.")
    
    logger.info(f"🤖 Generowanie AI podsumowania dla task_id: {data.task_id}")
    logger.info(f"💬 Prompt: '{data.prompt}'")
    logger.info(f"🔗 Include links: {data.include_links}")
    logger.info(f"📄 Fetch content: {data.fetch_content}")
    
    config = dfs_config.Configuration(username=DFS_LOGIN, password=DFS_PASSWORD)
    
    # Przygotowanie requestu
    request_data = [
        SerpAiSummaryRequestInfo(
            task_id=data.task_id,
            prompt=data.prompt,
            support_extra=data.support_extra,
            fetch_content=data.fetch_content,
            include_links=data.include_links
        )
    ]
    
    try:
        with dfs_api_provider.ApiClient(config) as api_client:
            api_instance = SerpApi(api_client)
            logger.debug("🤖 Wysyłanie żądania do DataForSEO AI Summary API...")
            
            # Wywołanie endpointu AI Summary
            api_response = api_instance.ai_summary(request_data)
            
            if not api_response.tasks:
                logger.warning("⚠️ Brak zadań w odpowiedzi API.")
                raise HTTPException(status_code=404, detail="Brak danych w odpowiedzi API.")
            
            task = api_response.tasks[0]
            
            if task.status_code != 20000:
                logger.error(f"❌ Błąd zadania: {task.status_code} - {task.status_message}")
                raise HTTPException(
                    status_code=400, 
                    detail=f"Błąd zadania DataForSEO: {task.status_message}"
                )
            
            if not task.result:
                logger.warning("⚠️ Brak wyników AI Summary.")
                raise HTTPException(
                    status_code=404, 
                    detail="Brak wyników AI Summary dla podanego task_id."
                )
            
            result = task.result[0]
            
            # Logowanie podstawowych informacji o wynikach
            logger.info(f"✅ Wygenerowano AI Summary")
            logger.info(f"📊 Liczba elementów: {result.items_count}")
            
            if result.items and len(result.items) > 0:
                summary_length = len(result.items[0].summary) if result.items[0].summary else 0
                logger.info(f"📝 Długość podsumowania: {summary_length} znaków")
            
            return {
                "task_info": {
                    "id": task.id,
                    "status_code": task.status_code,
                    "status_message": task.status_message,
                    "cost": task.cost,
                    "execution_time": task.time
                },
                "result": result.to_dict()
            }
    
    except Exception as e:
        logger.exception("❌ Błąd podczas generowania AI Summary")
        raise HTTPException(
            status_code=500, 
            detail=f"Błąd serwera podczas generowania AI Summary:\n{str(e)}"
        )

@router.post("/serp/google/organic/live/advanced/with-ai-summary")
async def get_serp_with_ai_summary(data: SerpOrganicInput, ai_prompt: Optional[str] = "Analyze these search results and provide key insights"):
    """
    Pobiera wyniki SERP i automatycznie generuje AI podsumowanie.
    Kombinuje oba endpointy w jeden call.
    """
    logger.info(f"🔄 Pobieranie SERP + AI Summary dla: '{data.keyword}'")
    
    # 1. Najpierw pobierz wyniki SERP
    serp_response = await get_google_organic_serp_live_advanced(data)
    
    if "task_info" not in serp_response:
        raise HTTPException(status_code=500, detail="Brak task_id w odpowiedzi SERP")
    
    task_id = serp_response["task_info"]["id"]
    logger.info(f"📋 Otrzymano task_id: {task_id}")
    
    # 2. Teraz wygeneruj AI Summary
    ai_input = SerpAiSummaryInput(
        task_id=task_id,
        prompt=ai_prompt,
        include_links=True,
        fetch_content=False  # żeby było szybciej
    )
    
    try:
        ai_response = await get_serp_ai_summary(ai_input)
        
        # 3. Połącz wyniki
        combined_response = {
            "serp_data": serp_response,
            "ai_summary": ai_response,
            "combined_cost": serp_response["task_info"]["cost"] + ai_response["task_info"]["cost"]
        }
        
        logger.info(f"✅ Zakończono SERP + AI Summary. Łączny koszt: ${combined_response['combined_cost']}")
        
        return combined_response
        
    except Exception as e:
        logger.warning(f"⚠️ SERP pobrano pomyślnie, ale AI Summary nie udało się: {str(e)}")
        # Zwróć przynajmniej wyniki SERP
        return {
            "serp_data": serp_response,
            "ai_summary": None,
            "ai_summary_error": str(e),
            "combined_cost": serp_response["task_info"]["cost"]
        }

@router.post("/serp/google/organic/live/advanced/filtered")
async def get_filtered_serp_data(data: SerpOrganicInput, element_types: list[str]):
    """
    Pobiera dane SERP i filtruje tylko wybrane typy elementów.
    
    Przykładowe typy elementów:
    - organic: organiczne wyniki wyszukiwania
    - paid: płatne reklamy
    - featured_snippet: wyróżnione fragmenty
    - people_also_ask: ludzie pytają też
    - knowledge_graph: graf wiedzy
    - local_pack: wyniki lokalne
    - images: obrazy
    - shopping: zakupy
    - top_stories: najważniejsze wiadomości
    """
    # Wywołaj główny endpoint
    full_response = await get_google_organic_serp_live_advanced(data)
    
    # Filtruj elementy według typu
    if 'result' in full_response and 'items' in full_response['result']:
        filtered_items = [
            item for item in full_response['result']['items'] 
            if item.get('type') in element_types
        ]
        
        full_response['result']['items'] = filtered_items
        full_response['result']['items_count'] = len(filtered_items)
        
        logger.info(f"🔍 Przefiltrowano do {len(filtered_items)} elementów typów: {', '.join(element_types)}")
    
    return full_response
